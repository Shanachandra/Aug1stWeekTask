function isPrime(n)
{
    var prime =true;
    if(n===2)
    {
        return true;
    }
    for( var i=2;i<n;i++)
    {
        if(n%i===0)
        {
            prime=false;
            return false;
        }
    }
    if(prime===true)
    {
        return true;
    }
 
}

function sumCSV(s)
{
    var sum=0;
    var splitted=s.split(",");
    for(var i=0;i<splitted.length;i++)
    {
        sum=sum+(+splitted[i]);
        
    }
    return sum;
  }
